    params = Array::slice.call process.argv, 2

        switch p
            when "-v", "--verbose"
                options.verbose = true
            when "-d", "--delete"
                options.removeDuplicates = true
            when "-f", "--fast"
                options.fast = true
            when "-s", "--superfast"
                options.superfast = true
            when "-o", "--output"
                options.output = true
            when "-h", "--help"
                showHelp()
                process.exit 0
            else
                folders.push p

        callback null if callback?
    catch ex
